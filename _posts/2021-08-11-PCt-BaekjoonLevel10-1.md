def s(N,A):
    T=3**N
    B=range(int(T/3),int(2*T/3))
    C=[i for i in range(len(A)) if i%T in B]
    for X in C:
        for Y in C:
            A[X][Y]=' '
    N-=1
    if N!=0:
        return s(N,A)
    else:
        return A

n=int(input())
a=[['*' for c in range(n)] for r in range(n)]
b=s(int(n**(1/3)),a)

p=''
for i in range(len(b)):
